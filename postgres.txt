CREATE DATABASE data_tool;
CREATE USER data_tool_user WITH PASSWORD 'Zz@12345';
ALTER ROLE data_tool_user SET client_encoding TO 'utf8';
ALTER ROLE data_tool_user SET default_transaction_isolation TO 'read committed';
ALTER ROLE data_tool_user SET timezone TO 'UTC';
GRANT ALL PRIVILEGES ON DATABASE data_tool TO data_tool_user;
ALTER USER data_tool_user CREATEDB;

CREATE TABLE documents
(
    id bigint NOT NULL GENERATED BY DEFAULT AS IDENTITY,
    is_active boolean NOT NULL,
    old_id bigint,
    date_start timestamp with time zone,
    date_end timestamp with time zone,
    created_at timestamp with time zone NOT NULL,
    created_user character varying(150) COLLATE pg_catalog."default" NOT NULL,
    changed_at timestamp with time zone NOT NULL,
    changed_user character varying(150) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT documents_pkey PRIMARY KEY (id)
)
